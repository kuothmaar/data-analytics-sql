    ID INT PRIMARY KEY,
    Product_Name VARCHAR(100),
    Category VARCHAR(50),
    Sales INT,

-- 2. Write an SQL query to find the total profit by category. Only include categories where total profit is greater than 500.
SELECT category, SUM(profit)
FROM Products
GROUP BY category
HAVING SUM(profit) >500;


-----Question 3:
Write an SQL query to find total sales per country. Only include countries with
total sales above 2500.

SELECT Country, SUM(Sales) AS total_sales
FROM Products 
GROUP BY Country
HAVING SUM(Sales) > 2500;


----Question 4:
Write an SQL query to find the average profit per category. Only show
categories with an average profit greater than 200.


SELECT category,AVG(profit)
FROM products
GROUP BY category
HAVING AVG(profit)>200;


---Question 5:
Write an SQL query to count the number of products per country. Only include countries with more than 2 products.



SELECT Country, COUNT(DISTINCT Product_Name) AS total_products
FROM Products
GROUP BY Country
HAVING COUNT(Product_Name);
